CREATE TABLE IF NOT EXISTS users (
    username VARCHAR(32) UNIQUE NOT NULL,
    email VARCHAR(32) UNIQUE NOT NULL,
    firstname VARCHAR(32) NOT NULL,
    lastname VARCHAR(32) NOT NULL,
    password VARCHAR(128) NOT NULL,
    accounttype INTEGER(3) NOT NULL,
    id INTEGER PRIMARY KEY AUTOINCREMENT UNIQUE NOT NULL
);

CREATE TABLE IF NOT EXISTS questions (
    category VARCHAR(128) NOT NULL,
    title VARCHAR(128) NOT NULL,
    question VARCHAR(512) NOT NULL,
    timeofquestion DATETIME NOT NULL,
    duplicate INTEGER(28) NOT NULL,
    id INTEGER PRIMARY KEY AUTOINCREMENT UNIQUE NOT NULL,
);

CREATE TABLE IF NOT EXISTS answers (
    response VARCHAR(512) NOT NULL,
    vote INTEGER(12),
    timeofanswer DATETIME NOT NULL,
    qId INTEGER PRIMARY KEY AUTOINCREMENT UNIQUE NOT NULL
);
